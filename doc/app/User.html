def last_order
  if orders.count == 0
    order = Order.new
    order.user = self
    order.save
    return order
  else
    if orders.last.done? || orders.last.delivery? || orders.last.failed?
      order = Order.new
      order.user = self
      order.save
      return order
    else
      return orders.last
    end
  end
end
